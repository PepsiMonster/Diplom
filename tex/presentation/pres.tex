\section{Введение и постановка задачи}

\begin{frame}{Актуальность исследования}
    \begin{itemize}
        \item Исследование ресурсных систем массового обслуживания (модель $GI/G/N/N/R$).
        \item \textbf{Проблема:} Вычислительная сложность точного аналитического расчета многомерных систем при немарковских предположениях.
        \item \textbf{Цель:} Анализ чувствительности стационарных характеристик.
    \end{itemize}

    \begin{block}{Ключевые метрики}
        В фокусе внимания находится вероятность потери (блокировки) заявки:
        \[
        P_{\text{loss}} = 1 - \frac{\lambda_{\text{accepted}}}{\lambda_{\text{total}}}
        \]
    \end{block}
\end{frame}
